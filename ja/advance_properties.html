#   commonColumnSetupBeforeDeleteInterceptorLogicMap: (NotNull - map-property or empty(map:{}))

#           ; U_USER=entity.getRStaff() \
#         }
#     ex) map:{ \
#           U_USER=dbflute.ldb.nongenerate.LdUserContext.getLoginUser() \
#         }
#
torque.commonColumnMap = map:{R_TIME=TIMESTAMP;U_TIME=TIMESTAMP;R_USER=VARCHAR;U_USER=VARCHAR}

torque.commonColumnSetupBeforeInsertInterceptorLogicMap = map:{ \

        ; U_USER=entity.getRUser() \
    }

        U_USER=dbflute.ldb.nongenerate.LdUserContext.getLoginUser() \
    }

torque.commonColumnSetupBeforeDeleteInterceptorLogicMap = map:{}

# _/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/
#                                                     Select Param
#                                                     _/_/_/_/_/_/

# /---------------------------------------------------------------------------
# [SelectQueryTimeout] (NotRequired - Default '-1')
torque.selectQueryTimeout = -1
# ----------------/

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    時には、「30分掛かってもいいから検索結果が帰って来て欲しい！」という画面やBatch処理があるやもしれません。
    その場合には、SQLのTimeout設定を延ばす必要があるため、この設定をMilli秒単位で指定します。

    素のS2Daoを利用すると、PreparedStatementが隠蔽されているため、QueryTimeoutを設定するのは
    結構大変です(独自拡張が必要)。DBFluteはS2DaoのSelectDynamicCommandを拡張しているため、
    この設定が容易に可能です。

    作者は実際にCatalog系DBのAdmin画面にてこの設定が必要となり利用しました。

# _/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/
#                                            Additional ForeignKey
#                                                     _/_/_/_/_/_/

# /---------------------------------------------------------------------------
# [Additional ForeignKey]
#   additionalForeignKeyMap: (NotRequired - Default 'map:{}')
#
#   If foreign key does not exist in your database,
#   you can set up here as virtual foreign key for DBFlute.
# 
#    ex) torque.additionalForeignKeyMap = map:{ \
#            ; FK_BOOK_AUTHOR = map:{ \
#                ; localTableName  = BOOK      ; foreignTableName  = AUTHOR \
#                ; localColumnName = AUTHOR_ID ; foreignColumnName = AUTHOR_ID \
#            } \
#            ; FK_BOOK_PUBLISHER = map:{ \
#                ; localTableName  = BOOK         ; foreignTableName  = PUBLISHER \
#                ; localColumnName = PUBLISHER_ID ; foreignColumnName = PUBLISHER_ID \
#            } \
#        }
#
torque.additionalForeignKeyMap = map:{ \
            ; FK_MULTIPLE_FK_TEST_LENDING_COLLECTION  = map:{ \
                ; localTableName  = LENDING_COLLECTION ; foreignTableName  = LIBRARY_USER \
                ; localColumnName = LIBRARY_ID/LB_USER_ID ; foreignColumnName = LIBRARY_ID/LB_USER_ID \
            } \
    }
# ----------------/

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    DBに物理的なForeignKey制約が無い場合にDBFluteに擬似的にFKを認識させる機能です。
    ProjectでFK制約をDBに全く貼らないProjectや、Viewに対して結合したい場合などに有効です。

    ＜複合FK＞
    複合列によるFKの場合は、'/'で区切ってColumn名を設定します。

    ＜localColumnNameの省略＞
    localColumnName を省略した場合は、foreignColumnNameと同じ名前のLocalTableのColumnが、
    localColumnNameとして利用されます。

    ＜foreignColumnNameの省略＞
    foreignColumnName を省略した場合は、ForeignTableのPrimaryKeyのColumnが、
    foreignColumnNameとして利用されます。

    ＜WildCard＞
    「localTableName=*」とすると指定されたlocalColumnNameが定義されている全てのTableに擬似FKを付与します。
    共通列などでFK制約が貼られていない場合などに有効です。
    (localColumnNameを省略した場合はforeignColumnNameと同じ名前で探しに行きます)

    注：
    細かいことを差し置いて極端なことを言うと、
    DBFluteにとってFKは「left outer joinするメソッドを生成するか否か」の要素です。
    この'Additional ForeignKey'を設定したからと言って、DBの本物のFK制約のように値の整合性Checkを
    するわけではありませんのでご注意下さい。
    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -



# _/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/
#                                                 Source Reduction
#                                                         _/_/_/_/
# /---------------------------------------------------------------------------
# [Make Deprecated] (NotRequired - Default 'true')
#   [true]
#     Make deprecated method and class and so on...
#
torque.isMakeDeprecated = false
# ----------------/

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    Deprecated指定のClassやMethodを作成するか否かを指定します。

    自動生成されるSourceの量を減らすためにも、DeprecatedのMethodをどうしても利用しなければ
    ならない理由が無い限り、この設定をfalseにすることをお奨めします。
    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -



# /---------------------------------------------------------------------------
# [Make BehaviorForUpdate] (NotRequired - Default 'true')
#   [true]
#     Make for-update method of behavior.
#
torque.isMakeBehaviorForUpdate = false
# ----------------/

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    Behaviorに定義されるSelectForUpdate関連のMethodを作成するか否かを指定します。

    「Project全体で更新Lockを掛けることはない」という方針の場合は、
    自動生成されるSourceの量を減らすためにも、この設定をfalseにすることをお奨めします。

    また、「時々ForUpdateすることがある」場合でも、ConditionBeanでForUpdateすることは可能なので、
    Behaviorに生成されるSelectForUpdate関連のMethodが不要であれば、遠慮なくfalseにしてOKです。
    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
