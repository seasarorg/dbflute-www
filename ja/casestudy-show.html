    /**
     * BOOK一覧検索。
     * 表示：BOOKの一部 + GENRE.GENRE_NAME + AUTHOR.AUTHOR_NAME
     * 条件：BOOKに対してBOOK_NAMEの'S2Dao'前方一致で、かつ、GENRE_CODEがnullでないもの
     */
    public void test_selectList_BookNameStartsWithS2Dao_and_GenreCode_is_not_null_Tx() {
        // ## Arrange ##
        _help.registerMyTableWithForeign();

        // ## Act ##
        final LdBookDao dao = (LdBookDao) getDaoSelector().getRDao(LdBookDao.class);
        final LdBookCB cb = new LdBookCB();
        cb.setupSelect_Genre();
        cb.setupSelect_Author();
        cb.query().setBookName_PrefixSearch("S2Dao");// '%'は内部で自動的に付与される
        cb.query().setGenreCode_IsNotNull();
        final java.util.List ls = dao.selectList(cb);
        assertListNotEmtpy(ls);

        // ## Assert ##
        final StringBuilder sb = new StringBuilder();
        sb.append(getLineSeparator());
        sb.append("-------------------------------------------------------------------------");
        for (LdBook book : ls) {
            sb.append(getLineSeparator());
            final BigDecimal bookId = book.getBookId();
            final String bookName = book.getBookName();
            if (!bookName.startsWith("S2Dao")) {
                fail("Wrong bookName: " + bookName);
            }
            final String genreCode = book.getGenreCode();
            assertNotNull(genreCode);
            final String genreName = book.getGenre().getGenreName();
            assertNotNull(genreName);
            final String authorName = book.getAuthor().getAuthorName();

            final String l = " | ";
            sb.append(bookId + l + bookName + l + genreName + l + authorName);
        }
        sb.append(getLineSeparator());
        sb.append("-------------------------------------------------------------------------");

        // ## Show ##
        _log.info(sb.toString());
    }

    /**
     * BOOK一覧検索。
     * 表示：BOOKの一部 + GENRE.GENRE_NAME
     * 条件：BOOKに対してGENRE_CODEが{'NOVEL', 'HOW-TO-BOOK'}のもの
     */
    public void test_selectList_GenreCode_in_NOVEL_and_HOW_TO_BOOK_Tx() {
        // ## Arrange ##
        _help.registerMyTableWithForeign();

        // ## Act ##
        final LdBookDao dao = (LdBookDao) getDaoSelector().getRDao(LdBookDao.class);
        final LdBookCB cb = new LdBookCB();
        cb.setupSelect_Genre();
        final String[] genreCodeArray = new String[] { "NOVEL", "HOW-TO-BOOK" };
        cb.query().setGenreCode_InScope(java.util.Arrays.asList(genreCodeArray));
        final java.util.List ls = dao.selectList(cb);

        // ## Assert ##
        final StringBuilder sb = new StringBuilder();
        sb.append(getLineSeparator());
        sb.append("-------------------------------------------------------------------------");
        for (LdBook book : ls) {
            sb.append(getLineSeparator());
            final BigDecimal bookId = book.getBookId();
            final String bookName = book.getBookName();
            final String genreCode = book.getGenreCode();
            assertNotNull(genreCode);
            if (!"NOVEL".equals(genreCode) && !"HOW-TO-BOOK".equals(genreCode)) {
                fail("Wrong genre code: " + genreCode);
            }
            final String genreName = book.getGenre().getGenreName();
            assertNotNull(genreName);

            final String l = " | ";
            sb.append(bookId + l + bookName + l + genreCode + l + genreName);
        }
        sb.append(getLineSeparator());
        sb.append("-------------------------------------------------------------------------");

        // ## Show ##
        _log.info(sb.toString());
    }
