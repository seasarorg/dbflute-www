    // Callback処理の作成
    final FileTokenizingCallback callback = new FileTokenizingCallback() {
        public void handleRowResource(FileTokenizingRowResource rowResource) {
            // nnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnn
            // 1行(1Row)読み込んだときの独自処理をここで実装する。
            // 1行(1Row)のTokenizeされた情報がRowResourceというObjectで利用できる。
            // 要件次第で変換してDBに保存したり、別のFileに出したり、などなど自由に実装すること。
            // - - - - - - - - - - - - - - - - - - 
            // Header情報を取得｛1行目のColumnName｝
            final FileTokenizingHeaderInfo headerInfo = rowResource.getFileTokenizingHeaderInfo();
            final java.util.List<String> columnNameList = headerInfo.getColumnNameList();

            // 値のListを取得｛Tokenizeされた1行分のData｝
            // これを利用して実装者は要件に沿った独自の処理を行う。
            final java.util.List<String> valueList = rowResource.getValueList();

            // Row文字列を取得｛Tokenizeされていないそのままの文字列｝
            // 通常はvalueListを利用して処理すればよいので必要はないが、
            // 特殊要件のために取得できるようにしている。
            final String rowString = rowResource.getRowString();

            // Row番号を取得
            // Header情報は考慮に入れずに、Data行として認識できるRowの番号。
            // 例えば、QuoteされたColumnのData内に改行が存在していて、
            // Editor上は2行になっていても、それは1行(1Row)とみなす。
            // このhandleRowResource()-Methodの呼び出された回数と一致する。
            // こちらも特殊要件のために取得できるようにしている。
            final int rowNumber = rowResource.getRowNumber();

            // Line番号を取得
            // 純粋に改行の数を考慮に入れた行番号。
            // InputしたFile上の行番号と一致する。(Editorが改行単位で行番号表示している場合に)
            // 処理に失敗した行をLogとして残しておくと運用者に優しいと思われる。
            final int lineNumber = rowResource.getLineNumber();
            // nnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnn
        }
    };

    // Optionの指定｛ここでは「Tab区切りでUTF-8のFile」であることを指定｝
    final FileTokenizingOption option = new FileTokenizingOption();
    option.delimitateByTab().encodeAsUTF8();

    // FileTokenの生成
    final FileToken fileToken = new FileTokenImpl();

    // 実行！ → Callback処理が1行(1Row)読み込み毎に呼び出される
    fileToken.tokenize("/tmp/tmp.tsv", callback, option);
