interface FileToken
    void tokenize(String filename
                , FileTokenizingCallback fileTokenizingCallback
                , FileTokenizingOption fileTokenizingOption
                ) throws java.io.FileNotFoundException, java.io.IOException;

    /- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    // 【Callback処理】の作成
    final FileTokenizingCallback callback = new FileTokenizingCallback() {
        public void handleRowResource(FileTokenizingRowResource rowResource) {
            // nnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnn
            // 1行(1Row)読み込んだときの独自処理をここで実装する。
            // 1行(1Row)のTokenizeされた情報がRowResourceというObjectで利用できる。
            // 要件次第で変換してDBに保存したり、別のFileに出したり、などなど自由に実装すること。
            // - - - - - - - - - - - - - - - - - - 
            // Header情報を取得｛1行目のColumnName｝
            final FileTokenizingHeaderInfo headerInfo = rowResource.getFileTokenizingHeaderInfo();
            final java.util.List<String> columnNameList = headerInfo.getColumnNameList();

            // 値のListを取得｛Tokenizeされた1行分のData｝
            // これを利用して実装者は要件に沿った独自の処理を行う。
            final java.util.List<String> valueList = rowResource.getValueList();

            // Row文字列を取得｛Tokenizeされていないそのままの文字列｝
            // 通常はvalueListを利用して処理すればよいので必要はないが、
            // 特殊要件のために取得できるようにしている。
            final String rowString = rowResource.getRowString();

            // Row番号を取得
            // Header情報は考慮に入れずに、Data行として認識できるRowの番号。
            // 例えば、QuoteされたColumnのData内に改行が存在していて、
            // Editor上は2行になっていても、それは1行(1Row)とみなす。
            // このhandleRowResource()-Methodの呼び出された回数と一致する。
            // こちらも特殊要件のために取得できるようにしている。
            final int rowNumber = rowResource.getRowNumber();

            // Line番号を取得
            // 純粋に改行の数を考慮に入れた行番号。
            // InputしたFile上の行番号と一致する。(Editorが改行単位で行番号表示している場合に)
            // 処理に失敗した行をLogとして残しておくと運用者に優しいと思われる。
            final int lineNumber = rowResource.getLineNumber();
            // nnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnnn
        }
    };

    // 【Optionの指定】｛ここでは「Tab区切りでUTF-8のFile」であることを指定｝
    final FileTokenizingOption option = new FileTokenizingOption();
    option.delimitateByTab().encodeAsUTF8();

    // 【FileTokenの生成】
    // StatelessなClassなので、Diconに登録してSingletonにしても構わない。
    final FileToken fileToken = new FileTokenImpl();

    // 【実行！】 → Callback処理が1行(1Row)読み込み毎に呼び出される
    fileToken.tokenize("/tmp/tmp.tsv", callback, option);

この機能を利用した処理が既にBehaviorの中に存在します。

interface BehaviorWritable
    TokenFileReflectionResult reflectTokenFile(final String filename
                                             , final TokenFileReflectionOption tokenFileReflectionOption
                                             ) throws java.io.FileNotFoundException, java.io.IOException;

このMethodは、指定されたTSV/CSVを読み込んで該当Tableに反映処理(createOrUpdate)を行います。
TSV/CSVが既に該当Tableの構造と一致する場合に、File名とOptionを指定するだけで取り込むことが可能です。
(Header情報として1行目にColumn名が存在することが前提)

    /- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    assertStringNotNullAndNotTrimmedEmpty("filename", filename);
    assertObjectNotNull("tokenFileReflectionOption", tokenFileReflectionOption);

    final String encoding = tokenFileReflectionOption.getEncoding();
    final String delimiter = tokenFileReflectionOption.getDelimiter();
    assertStringNotNullAndNotTrimmedEmpty("encoding", encoding);
    assertObjectNotNull("delimiter", delimiter);

    // Initialize result object!
    final TokenFileReflectionResult result = new TokenFileReflectionResult();
    final java.util.List<TokenFileReflectionFailure> failureList
                             = new java.util.ArrayList<TokenFileReflectionFailure>();
    result.setTokenFileReflectionFailureList(failureList);

    final FileTokenizingCallback callback = new FileTokenizingCallback() {
        public void handleRowResource(FileTokenizingRowResource rowResource) {
            final FileTokenizingHeaderInfo headerInfo = rowResource.getFileTokenizingHeaderInfo();
            final java.util.List<String> columnNameList = headerInfo.getColumnNameList();
            final java.util.List<String> valueList = rowResource.getValueList();

            // Set up columnNameList of result object.
            if (result.getColumnNameList() == null) {
                result.setColumnNameList(columnNameList);
            }

            Entity entity = null;
            try {
                // Create entity by the list of value composed of String.
                entity = createEntityByStringValueList(columnNameList, valueList);

                // Create or modify as non-strict.
                doCreateOrUpdateNonstrict(entity);

                // Increment successCount of result object.
                result.incrementSuccessCount();
            } catch (RuntimeException e) {
                if (tokenFileReflectionOption.isInterruptIfError()) {
                    throw e;
                }
                final TokenFileReflectionFailure failure = new TokenFileReflectionFailure();
                failure.setColumnNameList(columnNameList);
                failure.setValueList(valueList);
                failure.setRowString(rowResource.getRowString());
                failure.setRowNumber(rowResource.getRowNumber());
                failure.setLineNumber(rowResource.getLineNumber());
                if (entity != null) {
                    failure.setEntity(entity);
                }
                failure.setException(e);
                failureList.add(failure);
            }
        }
    };
    final FileTokenizingOption option = new FileTokenizingOption();
    option.setEncoding(encoding);
    option.setDelimiter(delimiter);
    final FileToken fileToken = new FileTokenImpl();
    fileToken.tokenize(filename, callback, option);
    return result;
