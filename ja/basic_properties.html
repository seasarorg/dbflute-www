# /---------------------------------------------------------------------------
# [JavaNameOfTableSameAsDbName] (NotRequired - Default 'false')
#   [true]
#     Java name of table is same as db name.
#       ex) If the table name is 'OrderDetail', then the java name is 'OrderDetail'.
#
#   [false]
#       ex) If the table name is 'ORDER_DETAIL', then the java name is 'OrderDetail'.
#       ex) If the table name is 'OrderDetail', then the java name is 'Orderdetail'.
#
torque.isJavaNameOfTableSameAsDbName = false
# ----------------/

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    Table名をClass名などで利用するときの命名Ruleです。

    Table名が'ORDER_DETAIL' or 'order_detail'というような'_'繋がりの場合は、falseでOKです。(消してOKです)
    Table名が'OrderDetail'というようなClass名で利用したい名前と同じ場合は、trueでOKです。

    時々、'mst_OrderDetail'という場合のProjectがあります。(作者が体験)
    その場合はこのPropertyをfalseにして、以下のPropertyを指定します。

      torque.isAvailableToLowerInGeneratorUnderscoreMethod = false

    すると、'mst_OrderDetail'は、'MstOrderdetail'ではなく、'MstOrderDetail'となります。
    但し注意しなけらばならないのは、そのProjectにもし'mst_SALES_REPORT'というTableが混在していたら、
    このTableのClass名は、'MstSALESREPORT'になってしまいます。
    (つまり、上記の isAvailableToLowerInGeneratorUnderscoreMethod は、
     単語の2文字目以降をtoLowerするか否かを示します)

    ※極力、DatabaseのTable名は規則正しいものでお願いします。
    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

# _/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/
#                                                    ProjectPrefix
#                                                   _/_/_/_/_/_/_/

# /---------------------------------------------------------------------------
# [PrefixAndSuffix]
#
#   projectPrefix: (NotRequired - Default null)
#     If the property is 'Ld', all class names are 'LdXxx'.
#
#   isAppendProjectSuffixToComponentName: (NotRequired - Default 'true')
#     Default value is true.
#     [true]
#       The component name of bhv and dao have the prefix of project.
#
#     [false]
#       Non.
#
torque.projectPrefix		= Ld
torque.isAppendProjectSuffixToComponentName = true
# ----------------/

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    [projectPrefix]は、Class名の前後に付加する文字列の指定です。

    {projectPrefix}は、すべてのClass名の最初に「XX」を付けたいという場合に利用します。
    Projectなどで、統一される場合に有効です。そうでない場合は未指定でOKです。

    また、DBFluteで複数DB対応する場合に有効です。
    複数DB対応する場合は、RootPackageを変えた２つのDBFluteClass群を作成します。
    その場合、互いのallcommon内に同名Classが作成されてしまいますので、
    互いに別のprojectPrefixを付けることで明確に違いを付ける事ができます。

    など、DBFlute-0.4.4以降では、このprojectPrefixはdbflute.diconに登録される
    BehaviorのComponent名にも付与されます。

    【注意】
    InnerClassに関しては、このPrefixは付与されません。


    [isAppendProjectSuffixToComponentName]は、DBFlute-0.4.3 以前を利用されてして、
    projectPrefixがdbflute.diconに登録されるBehaviorのComponent名にも付与されたくない方の
    ためのPropertyです。falseにすることによって、DBFlute-0.4.3 以前と同様の判定になります。

    詳しくはこちら↓
    DBFlute-0.4.3 から 0.4.4 への移行時の注意

# /---------------------------------------------------------------------------
# [DBFluteDicon Information]
#   dbfluteDiconNamespace: (NotRequired - Default 'dbflute')
#     The namespace of dbflute dicon.
#
#   dbfluteDiconPackageName: (NotRequired - Default '')
#     The package name of dbflute dicon.
#
#   dbfluteDiconFileName: (NotRequired - Default 'dbflute.dicon')
#     The file name of dbflute dicon. (Required)
#
#   j2eeDiconResourceName: (NotRequired - Default Java:'j2ee.dicon' C#:'ado.dicon')
#     The resource name of J2EE dicon.

#

#     The map of the original dbflute component.
#     If componentName have '*' as prefix, the component should be Dao.

#             ; *bookAuthorDao = map:{ className=dbflute.ldb.nongenerate.dao.BookAuthorDao } \
#             ; blackListHandler = map:{ className=dbflute.ldb.nongenerate.handler.BlackListHandlerImpl } \
#         }
#
#   originalBehaviorAspectMap: (NotRequired - Default 'map:{}')
#     The map of the original behavior aspect.
#     ex) map:{ interceptorName = map:{ \
#                   className=org.seasar.dbflute.InterceptorClassName \
#                   ; pointcut=selectList.*, selectEntity.* \
#               } \
#         }
#
#   isAvailableBehaviorRequiresNewTx: (NotRequired - Default true)
#     Is available the behavior setting about requiresNewTx in dbflute.dicon?
#
#   isAvailableBehaviorRequiredTx: (NotRequired - Default true)
#     Is available the behavior setting about requiredTx in dbflute.dicon?
#
#
#   *Attension:
#      It is necessary that main dicon of application includes dbflute dicon.
#      ex) App dicon includes dbflute dicon.
#
torque.dbfluteDiconNamespace        = dbflute
torque.dbfluteDiconPackageName      = ../resources
torque.dbfluteDiconFileName         = dbflute-ldb.dicon

torque.requiredTxComponentName      = requiredTx
torque.requiresNewTxComponentName   = requiresNewTx
torque.dbfluteDiconBeforeJ2eeIncludeDefinitionMap = map:{ dbflute/jdbc-xxx.dicon = dummy }
torque.dbfluteDiconOtherIncludeDefinitionMap = map:{ dbflute/common.dicon = dummy }
torque.originalDBFluteComponentMap = map:{ \
             ; *libraryDetailDao = map:{ className=dbflute.ldb.nongenerate.dao.LibraryDetailDao } \
             ; blackListHandler  = map:{ className=dbflute.ldb.nongenerate.handler.BlackListHandlerImpl } \
         }
torque.originalBehaviorAspectMap = map:{}
torque.isAvailableBehaviorRequiresNewTx = true
torque.isAvailableBehaviorRequiredTx = true

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    diconの設定に関するPropertyです。

    {dbfluteDiconNamespace}は、dbflute.diconのnamespaceです。
    Default値の「dbflute」で問題なければ未指定で構いません。

      ※daoDiconNamespaceと記述しても永遠に同じ動作をします。
        同時に存在する場合はdbfluteDiconNamespaceが優先されます。


    {dbfluteDiconPackageName}は、dbflute.diconの出力先です。
    Default値の「Sourceの出力先Directory直下(src/main/javaなど)」で問題なければ未指定で構いません。
    もし、別の場所に出力させたいのであれば「../resources」というように相対Pathで指定します。
    (上記のLDBの場合は、一般的なdiconの配置先src/main/resourcesに出力されるように指定しています)
    また、Test用Projectなどが存在して、もう一つ全く同じdbflute.diconを別の場所に出力したいという場合は、
    ';'で区切ってもう一つPathを指定することで可能です。ex) ../resources; ../../test/resources

      ※daoDiconPackageNameと記述しても永遠に同じ動作をします。
        同時に存在する場合はdbfluteDiconPackageNameが優先されます。


    {dbfluteDiconFileName}は、dbflute.diconのFile名です。
    Default値の「dbflute.dicon」で問題なければ未指定で構いません。

        - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 
        DBFlute-0.3.2まではDefault値が'dao.dicon'でした。
        dbflute.diconに変更した理由は、'dao.dicon'という名前のResourceが
        既にS2DaoのJarの中に含まれており、Projectを細かく分けて参照した際に
        誤動作する可能性があるからです(S2Dao.jar内のdao.diconを読んでしまう)。
        - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 

      ※daoDiconFileNameと記述しても永遠に同じ動作をします。
        同時に存在する場合はdbfluteDiconFileNameが優先されます。


    {j2eeDiconResourceName}は、j2ee.diconのResource名です。
    dao.diconの中で宣言されるj2ee.diconへのinclude-pathとなります。
    Default値の「j2ee.dicon」で問題なければ未指定で構いません。

      ※j2ee.diconは、DBFluteは自動生成対象外です。各Projectにて準備して下さい。
        S2Containerのjar内部に存在するj2ee.diconを利用する場合は、j2ee.diconが参照するjdbc.diconを
        用意してください。

        - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 
        DBFlute-0.3.2まではDefault値が'j2ee.dicon'でした。
        また、j2eeDiconResourceNameはdeprecateされました(書けば正常に動作します)。
        詳しくはこちら → DBFlute-0.3.2 to 0.4.0
        - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 

	
	DBFlute-0.4.2以降について
	　　★★★★★★★★★★★★★★★★★★★★★★★★★★★★★★★
	　　注意点：
	　　　[DBFLUTE-51]の「j2eeDiconResourceのDefault値をj2ee.diconに戻す。」
	　　　対象：j2eeDiconResourceをbuild.propertiesに指定せずにデフォルト値の「jdbc.dicon」を利用されている方
	　　　作者が勘違いして、jdbc.diconをデフォルト値にとるようにしてしまいましたが、
	　　　S2の標準では Seasarのjarの内部に j2ee.dicon が存在していて、それが jdbc.dicon をincludeしているため、
	　　　基本、dbflute.diconからはj2ee.diconをincludeすれば独自で作成したjdbc.diconが読み込まれることになります。
	　　　S2の標準に従うために、真に申し訳御座いませんが、デフォルト値を「j2ee.dicon」に戻させて頂きます。
	　　　　→ 参考 - [Seasar-user:6168] DBFluteの０．４．０以降の設定について
	　　　　→ 参考 - こちら

	　　　　＜S2Containerのjar内部に存在するj2ee.diconを利用できる方＞

	　　　　　 お手数ですが、build.propertiesにj2eeDiconResourceNameプロパティで明示的に「jdbc.dicon」とするか、
	　　　　　 既存の「jdbc.dicon」を「j2ee.dicon」に名前を変更するようにお願いします。
	　　
	　　ご迷惑お掛けしました。
	　　
	　　今後、j2eeDiconResourceNameでもjdbcDiconResourceNameでもどちらでも指定しても永遠に同じ動作をします。
	　　★★★★★★★★★★★★★★★★★★★★★★★★★★★★★★★
	


    {requiredTxComponentName}は、jdbc.diconで定義されるrequiredTxのComponent名です。
    Default値の「Java:requiredTx/C#:LocalRequiredTx」で問題なければ未指定で構いません。


    {requiresNewTxComponentName}は、jdbc.diconで定義されるrequiresNewTxのComponent名です。
    Default値の「Java:requiresNewTx/C#:LocalRequiresNewTx」で問題なければ未指定で構いません。


    {dbfluteDiconBeforeJ2eeIncludeDefinitionMap}は、dbflute.diconでj2ee.dicon以外の別のDiconを
    Includeしたい場合に利用します。
    dbflute.diconは自動生成なので、手動で修正しても後に上書きされてしまうため、このPropertyが存在します。
    特に何もなければ未指定で構いません。
    j2ee.diconの宣言よりも下の行に出力されるため、S2ContainerがComponentの探索を行う際に
    S2Containerの動作仕様によりj2ee.diconよりも後に評価されます。
    mapのKey値にdbflute.diconにincludeされる(されたい)diconのResourceNameを記述します。
    Value値は利用していませんので'dummy'とお書き下さい。(将来の拡張領域の予定)

      ※daoDiconBeforeJ2eeIncludeDefinitionMapと記述しても永遠に同じ動作をします。
        同時に存在する場合はdbfluteDiconBeforeJ2eeIncludeDefinitionMapが優先されます。

      ※S2ContainerのXInclude機能を利用することにより、このPropertyを利用しなくても同様の動きを
        実現することが可能です。XIncludeはComponentだけをかき集めますので、「include部分だけをOverrideする」
        ようなことが可能になります。詳しくはこちら
        (但し、Xerces-2.7.0 以上が必須であることにご注意下さい)


    {dbfluteDiconOtherIncludeDefinitionMap}は、dbflute.diconでj2ee.dicon以外の別のDiconを
    Includeしたい場合に利用します。
    dbflute.diconは自動生成なので、手動で修正しても後に上書きされてしまうため、このPropertyが存在します。
    特に何もなければ未指定で構いません。
    j2ee.diconの宣言よりも下の行に出力されるため、S2ContainerがComponentの探索を行う際に
    S2Containerの動作仕様によりj2ee.diconよりも後に評価されます。
    mapのKey値にdbflute.diconにincludeされる(されたい)diconのResourceNameを記述します。
    Value値は利用していませんので'dummy'とお書き下さい。(将来の拡張領域の予定)

      ※daoDiconOtherIncludeDefinitionMapと記述しても永遠に同じ動作をします。
        同時に存在する場合はdbfluteDiconOtherIncludeDefinitionMapが優先されます。

      ※dbfluteDiconBeforeJ2eeIncludeDefinitionMapと同様にS2ContainerのXInclude機能の利用で代替できます。


    {originalDBFluteComponentMap}は、dbflute.diconに独自のComponentを登録したい場合に利用します。
    dbflute.diconは自動生成なので、手動で修正しても後に上書きされてしまうため、このPropertyが存在します。
    手動で作成されたDaoはこれを利用して登録します。ComponentNameにPrefixとして'*'を付けるとDaoとして認識されます。
    特に何もなければ未指定で構いません。

      ※originalDaoComponentMapと記述しても永遠に同じ動作をします。
        同時に存在する場合はoriginalDBFluteComponentMapが優先されます。

      ※dbfluteDiconBeforeJ2eeIncludeDefinitionMapと同様にS2ContainerのXInclude機能の利用で代替できます。


    {originalBehaviorAspectMap}は、dbflute.diconに独自のBehaviorAspect(Interceptor)を登録したい場合に利用します。
    dao.diconは自動生成なので、手動で修正しても後に上書きされてしまうため、このPropertyが存在します。
    特に何もなければ未指定で構いません。

    {isAvailableBehaviorRequiresNewTx}は、dbflute.diconに登録されているBehaviorの
    requiresNewTxの設定を有効にするか否かを判定します。
    ※DBFlute-0.4.4から

    {isAvailableBehaviorRequiredTx}は、dbflute.diconに登録されているBehaviorの
    requiredTxの設定を有効にするか否かを判定します。
    ※DBFlute-0.4.4から


    ※Applicationが利用するxxx.diconがdbflute.diconをIncludeする必要があります。
      例えば、app.diconがdbflute.diconをincludeするのが一般的です。

    自動生成されるClassのPackage名関連の設定です。

    {packageBase}は、すべてのPackage設定のBaseのPathとなります。
    他のそれぞれのPackage設定にはDefault値がありますので、それで問題なければこのPropertyのみ設定でOKです。

    【C#版ご利用の方へ注意】
    DBFluteの基本方針は、Package名は小文字始まりです。
    申し訳ありませんが、C#版においてもそのようになります。
    ここのPropertyで 'DBFlute.Ldb.Allcommon', 'DBFlute.Ldb.Bsbhv', '...'というように指定することで
    大文字始まりのNamespaceを実現することは可能ですが、AllcommonやBsdao以下のDBFluteが付加する
    cbean/customizeなどのNamespaceは小文字始まりとなります。
    将来的に修正されるかもしれませんが、現状(2006/12/30時点)ではこのような動作となります。
    ご了承下さい。

# Unnecessary tables: (NotNull - list:{TABLE_HINT;TABLE_HINT})
#   Default value is empty-list.
#
#   For example, if you want to except the table that the name starts with 'D_',
#   you should set this property as follows:

#

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    JDBC-Taskの情報取得対象の除外Tableの指定です。

    「このTableは自動生成しなくていい！」というTableがあったらこちらを指定します。
    たまーに、Database内部のSystemTableが取得できてしまうDatabaseがあります。
    そのときは、ここで指定して除外します。

    指定には、前方一致・後方一致・完全一致が利用可能です。
      前方一致 → prefix:         ex) D_で始まるTableを除外    → prefix:D_
      後方一致 → suffix:         ex) _TESTで終わるTableを除外 → suffix:_TEST

    JDBC-Taskの情報取得対象の対象Tableの指定です。

    「このTableとこのTableとこの...だけ自動生成したい！」という場合はこちらを指定します。

    ※基本的に、対象Tableを指定したときは除外Tableの設定は利用しません。
