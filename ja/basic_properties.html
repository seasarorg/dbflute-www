# /---------------------------------------------------------------------------
# [JavaNameOfTableSameAsDbName] (NotRequired - Default 'false')
#   [true]
#     Java name of table is same as db name.
#       ex) If the table name is 'OrderDetail', then the java name is 'OrderDetail'.
#
#   [false]
#       ex) If the table name is 'ORDER_DETAIL', then the java name is 'OrderDetail'.
#       ex) If the table name is 'OrderDetail', then the java name is 'Orderdetail'.
#
torque.isJavaNameOfTableSameAsDbName = false
# ----------------/

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    Table名をClass名などで利用するときの命名Ruleです。

    Table名が'ORDER_DETAIL' or 'order_detail'というような'_'繋がりの場合は、falseでOKです。(消してOKです)
    Table名が'OrderDetail'というようなClass名で利用したい名前と同じ場合は、trueでOKです。

    時々、'mst_OrderDetail'という場合のProjectがあります。(作者が体験)
    その場合はこのPropertyをfalseにして、以下のPropertyを指定します。

      torque.isAvailableToLowerInGeneratorUnderscoreMethod = false

    すると、'mst_OrderDetail'は、'MstOrderdetail'ではなく、'MstOrderDetail'となります。
    但し注意しなけらばならないのは、そのProjectにもし'mst_SALES_REPORT'というTableが混在していたら、
    このTableのClass名は、'MstSALESREPORT'になってしまいます。(つまり単語の2文字目以降をtoLowerするか否か)

    ※極力、DatabaseのTable名は規則正しいものでお願いします。
    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

# _/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/_/
#                                                            Dicon
#                                                             _/_/

# /---------------------------------------------------------------------------
# [DaoDicon Information]
#   daoDiconNamespace: (NotRequired - Default 'dbflute')
#     The namespace of dao dicon.
#
#   daoDiconPackageName: (NotRequired - Default '')
#     The package name of dao dicon.
#
#   daoDiconFileName: (NotRequired - Default 'dbflute.dicon')
#     The file name of dao dicon. (Required)
#
#   jdbcDiconResourceName: (NotRequired - Default 'jdbc.dicon')
#     The resource name of JDBC dicon.
#
#   requiredTxComponentName: (NotRequired - Default Java:'requiredTx' C#:'LocalRequiredTx')
#     The component name of 'requiredTx'-component.
#
#   requiresNewTxComponentName: (NotRequired - Default Java:'requiresNewTx' C#:'LocalRequiresNewTx')
#     The component name of 'requiresNewTx'-component.
#
#   daoDiconOtherIncludeDefinitionMap: (NotRequired - Default 'map:{}')
#     The definition map of dao-dicon's include path.
#     ex) map:{ dbflute/common.dicon = dummy }
#
#   originalDaoComponentMap: (NotRequired - Default 'map:{}')
#     The map of the original dao component.
#     If componentName have '*' as prefix, the component should be Dao.
#     ex) map:{ \
#             ; *bookAuthorDao = map:{ className=dbflute.ldb.nongenerate.dao.BookAuthorDao } \
#             ; blackListHandler = map:{ className=dbflute.ldb.nongenerate.handler.BlackListHandlerImpl } \
#         }
#
#   originalBehaviorAspectMap: (NotRequired - Default 'map:{}')
#     The map of the original behavior aspect.
#     ex) map:{ shopIdConditionAppendInterceptor = map:{ \
#                   className=MspadminBiz.Dao.Original.ShopIdConditionAppendInterceptor \
#                   ; pointcut=DelegateRead.*, DelegateSelect.* \
#               } \
#         }
#
#   *Attension:
#      It is necessary that main dicon of application includes dao dicon.
#
torque.daoDiconNamespace            = dbflute
torque.daoDiconPackageName          = ../resources
torque.daoDiconFileName             = dbflute-ldb.dicon
torque.jdbcDiconResourceName        = jdbc-ldb.dicon
torque.requiredTxComponentName      = requiredTx
torque.requiresNewTxComponentName   = requiresNewTx
torque.daoDiconOtherIncludeDefinitionMap = map:{}
torque.originalDaoComponentMap = map:{ \
             ; *libraryDetailDao = map:{ className=dbflute.ldb.nongenerate.dao.LibraryDetailDao } \
             ; blackListHandler  = map:{ className=dbflute.ldb.nongenerate.handler.BlackListHandlerImpl } \
         }
torque.originalBehaviorAspectMap = map:{}
# ----------------/

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    diconの設定に関するPropertyです。

    {daoDiconNamespace}は、dao.diconのnamespaceです。
    Default値の「dbflute」で問題なければ未指定で構いません。

    {daoDiconPackageName}は、dao.diconの出力先です。
    Default値の「Sourceの出力先Directory直下(src/main/javaなど)」で問題なければ未指定で構いません。
    もし、別の場所に出力させたいのであれば「../resources」というように相対Pathで指定します。
    (上記のLDBの場合は、一般的なdiconの配置先src/main/resourcesに出力されるように指定しています)

    {daoDiconFileName}は、dao.diconのFile名です。
    Default値の「dbflute.dicon」で問題なければ未指定で構いません。

        - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 
        DBFlute-0.3.2まではDefault値が'dao.dicon'でした。
        dbflute.diconに変更した理由は、'dao.dicon'という名前のResourceが
        既にS2DaoのJarの中に含まれており、Projectを細かく分けて参照した際に
        誤動作する可能性があるからです(S2Dao.jar内のdao.diconを読んでしまう)。
        - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 

    {jdbcDiconResourceName}は、jdbc.diconのResource名です。
    dao.diconの中で宣言されるjdbc.diconへのinclude-pathとなります。
    Default値の「jdbc.dicon」で問題なければ未指定で構いません。

      ※jdbc.diconは、DBFluteは自動生成対象外です。各Projectにて準備して下さい。
        その各Projectで準備されたjdbc.diconに対する参照をdao.diconに設定するためにこのPropertyが存在します。

        - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 
        DBFlute-0.3.2まではDefault値が'j2ee.dicon'でした。
        また、j2eeDiconResourceNameはdeprecateされました(書けば正常に動作します)。
        詳しくはこちら → DBFlute-0.3.2 to 0.4.0
        - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 

    {requiredTxComponentName}は、jdbc.diconで定義されるrequiredTxのComponent名です。
    Default値の「Java:requiredTx/C#:LocalRequiredTx」で問題なければ未指定で構いません。

    {requiresNewTxComponentName}は、jdbc.diconで定義されるrequiresNewTxのComponent名です。
    Default値の「Java:requiresNewTx/C#:LocalRequiresNewTx」で問題なければ未指定で構いません。

    {daoDiconOtherIncludeDefinitionMap}は、dao.diconでj2ee.dicon以外の別のDiconをIncludeしたい場合に利用します。
    dao.diconは自動生成なので、手動で修正しても後に上書きされてしまうため、このPropertyが存在します。
    特に何もなければ未指定で構いません。

    {originalDaoComponentMap}は、dao.diconに独自のComponentを登録したい場合に利用します。
    dao.diconは自動生成なので、手動で修正しても後に上書きされてしまうため、このPropertyが存在します。
    手動で作成されたDaoはこれを利用して登録します。ComponentNameにPrefixとして'*'を付けるとDaoとして認識されます。
    特に何もなければ未指定で構いません。

    {originalBehaviorAspectMap}は、dao.diconに独自のBehaviorAspect(Interceptor)を登録したい場合に利用します。
    dao.diconは自動生成なので、手動で修正しても後に上書きされてしまうため、このPropertyが存在します。
    特に何もなければ未指定で構いません。

    ※Applicationが利用するxxx.diconがdao.diconをIncludeする必要があります。
    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

# Unnecessary tables: (NotNull - list:{TABLE_HINT;TABLE_HINT})
#   Default value is empty-list.
#
#   For example, if you want to except the table that the name starts with 'D_',
#   you should set this property as follows:
#     ex) list:{prefix:D_}
#
#   And other example:
#     ex) list:{prefix:D_; suffix:_Test; sysobjects; dtproperties}
#
torque.table.except.list = list:{}

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    JDBC-Taskの情報取得対象の除外Tableの指定です。

    「このTableは自動生成しなくていい！」というTableがあったらこちらを指定します。
    たまーに、Database内部のSystemTableが取得できてしまうDatabaseがあります。
    そのときは、ここで指定して除外します。

    指定には、前方一致・後方一致・完全一致が利用可能です。
      前方一致 → prefix:         ex) D_で始まるTableを除外    → prefix:D_
      後方一致 → suffix:         ex) _TESTで終わるTableを除外 → suffix:_TEST
    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -



# Target tables: (NotNull - list:{TABLE_HINT;TABLE_HINT})
#   Default value is empty-list(All tables are target).
#   If this property is empty-list, the property of [Unnecessary tables] is valid.
#
#   For example, if you want to contains the table that the name starts with 'D_',
#   you should set this property as follows:
#     ex) list:{prefix:D_}
#
#   And other example:
#     ex) list:{prefix:D_; suffix:_Test; sysobjects; dtproperties}
#
#   If a table defined at this property is defined at [Unnecessary tables] too,
#   the table is target of generating. (ignoring [Unnecessary tables])
#

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    JDBC-Taskの情報取得対象の対象Tableの指定です。

    「このTableとこのTableとこの...だけ自動生成したい！」という場合はこちらを指定します。
    もし、あるTableが除外Tableで指定されていて、かつ対象Tableでも指定されている場合は、
    そのTableは自動生成されます。｛除外Tableの指定が無視される｝
    ※基本的に、対象Tableを指定したときは除外Tableの設定は利用しません。

# /---------------------------------------------------------------------------
# [S2Dao Version]
#   isVersionAfter1040: (NotRequired - Default 'true')
#     If your s2dao's version is greater than(or equal) S2Dao-1.0.40,
#     this property should be true.
#
#   *{Java Only}
#
torque.isVersionAfter1040 = true
# ----------------/

    - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
    Java版S2Dao-1.0.40以降を利用するか否かの指定です。

    S2Dao-1.0.40において、DaoMetaDataImplの構造が変わりました(拡張しやすいようにRefactoringされた)。
    それに伴って、DBFluteではS2Dao-1.0.40以降を利用しているのかどうかをどうしても判別しなければなりません。
    もし、どうしてもS2DaoのVersionをS2Dao-1.0.40以降に上げられない人は、false に指定して下さい。

    Defaultは{S2Dao-1.0.40以降}です。
